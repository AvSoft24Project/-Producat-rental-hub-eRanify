Create Table USERS(
	ID INT PRIMARY KEY,
	USER_NAME VARCHAR(100),
	USER_EMAIL_ID VARCHAR(100),
	USER_ADDRESS VARCHAR(255),
	USER_PHONE_NUMBER VARCHAR(20)
)

CREATE TABLE COMPLAINT(
	ID INT PRIMARY KEY,
	USERS_ID INT,
	COMPLAINT_TEXT TEXT,
	STATUS VARCHAR(50),
	PRIORITY VARCHAR(20),
	PHOTOS TEXT,
	FOREIGN KEY (USERS_ID)REFERENCES USERS(ID)

)
